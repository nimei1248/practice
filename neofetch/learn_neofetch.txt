## install neofetch
wget https://github.com/dylanaraps/neofetch/archive/2.0.1.zip neofetch-2.0.1.zip
unzip neofetch-2.0.1.zip
cd neofetch-2.0.1
PREFIX=/usr/local make install

which neofetch

## 设置别名,使用默认的也可以
alias neofetch2="neofetch \
--config off \
--block_range 1 8 \
--bold off \
--uptime_shorthand on \
--gtk_shorthand on \
--colors 4 1 8 8 8 7 \
"

## 配置文件
\cp ~/.config/neofetch/config ~/.config/neofetch/config-20161216
